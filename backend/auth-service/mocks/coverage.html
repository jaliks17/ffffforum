package auth import ( "errors" "time" "github.com/dgrijalva/jwt-go" "golang.org/x/crypto/bcrypt" ) type Config struct { TokenSecret string TokenExpiration time.Duration } func GenerateToken(userID int64, role string, username string, secret string, expiration time.Duration) (string, error) { claims := jwt.MapClaims{ "user_id": userID, "role": role, "username": username, "exp": time.Now().Add(expiration).Unix(), } token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims) return token.SignedString([]byte(secret)) } func ParseToken(tokenString string, secret string) (*jwt.Token, error) { token, err := jwt.Parse(tokenString, func(token *jwt.Token) (interface{}, error) { if _, ok := token.Method.(*jwt.SigningMethodHMAC); !ok { return nil, errors.New("invalid signing method") } return []byte(secret), nil }) return token, err } func HashPassword(password string) (string, error) { hashedPassword, err := bcrypt.GenerateFromPassword([]byte(password), bcrypt.DefaultCost) if err != nil { return "", err } return string(hashedPassword), nil }
